Unveil.js

Entrance animations


    /* CSS */
    img {
      opacity: 0;
      transition: opacity .3s ease-in;
    }
    

    // Javascript
    $(document).ready(function() {

      $("img")
        .on("unveil", function() {
          $(this).load(function() {
            this.style.opacity = 1;
          });
        })
        .unveil();

    });
